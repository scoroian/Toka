# Tasks Feature (Spec-05) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement full task CRUD with recurrence engine, assignment rotation, task identity (icon/emoji), and the "All Tasks", "Task Detail", and "Create/Edit Task" screens.

**Architecture:** Domain models use freezed (sealed union for RecurrenceRule). Pure `RecurrenceCalculator` utility handles DST-aware next-due computation via the `timezone` package. `TasksRepositoryImpl` writes directly to Firestore (`homes/{homeId}/tasks/{taskId}`). Application layer uses Riverpod providers. Presentation follows the existing ConsumerWidget pattern.

**Tech Stack:** Flutter/Dart, Riverpod + riverpod_annotation, freezed, Firestore (fake_cloud_firestore for tests), timezone package (DST), mocktail.

```dart
// lib/features/tasks/domain/task_validator.dart
import '../../../features/homes/domain/home_limits.dart';
import 'task.dart';

sealed class ValidationResult {
  const ValidationResult();
  static const ValidationResult ok = _ValidationOk();
  static ValidationResult error(String code) => _ValidationError(code);

  bool get isOk => this is _ValidationOk;
  String? get errorCode =>
      this is _ValidationError ? (this as _ValidationError).code : null;
}

final class _ValidationOk extends ValidationResult {
  const _ValidationOk();
}

final class _ValidationError extends ValidationResult {
  const _ValidationError(this.code);
  final String code;
}

class TaskValidator {
  static ValidationResult validate(
    TaskInput input,
    HomeLimits limits,
    int currentActiveCount,
  ) {
    if (input.title.trim().isEmpty) {
      return ValidationResult.error('title_required');
    }
    if (input.title.length > 60) {
      return ValidationResult.error('title_too_long');
    }
    if (!limits.isPremium && currentActiveCount >= 4) {
      return ValidationResult.error('free_task_limit');
    }
    if (input.assignmentOrder.isEmpty) {
      return ValidationResult.error('no_assignees');
    }
    return ValidationResult.ok;
  }
}
```

```dart
// lib/core/utils/recurrence_calculator.dart
import 'package:timezone/timezone.dart' as tz;

import '../../features/tasks/domain/recurrence_rule.dart';

class RecurrenceCalculator {
  static int _weekdayFromString(String weekday) {
    const map = {
      'MON': DateTime.monday,
      'TUE': DateTime.tuesday,
      'WED': DateTime.wednesday,
      'THU': DateTime.thursday,
      'FRI': DateTime.friday,
      'SAT': DateTime.saturday,
      'SUN': DateTime.sunday,
    };
    return map[weekday.toUpperCase()] ?? DateTime.monday;
  }

  static ({int hour, int minute}) _parseTime(String time) {
    final parts = time.split(':');
    return (hour: int.parse(parts[0]), minute: int.parse(parts[1]));
  }

  /// Returns the last day of month [month] in year [year].
  static int _lastDayOfMonth(int year, int month) {
    return DateTime(year, month + 1, 0).day;
  }

  /// Find the date of the Nth weekday in a given year/month.
  /// Returns null if the Nth occurrence doesn't fit in the month.
  static DateTime? _nthWeekday(
      int year, int month, int weekOfMonth, int weekday) {
    final first = DateTime(year, month, 1);
    final daysUntilFirst = (weekday - first.weekday + 7) % 7;
    final firstOccurrence = first.add(Duration(days: daysUntilFirst));
    final nth = firstOccurrence.add(Duration(days: (weekOfMonth - 1) * 7));
    if (nth.month != month) return null;
    return nth;
  }

  static DateTime calculateNextDue(RecurrenceRule rule, DateTime from) {
    return rule.when(
      hourly: (every, startTime, endTime, timezone) =>
          _calculateHourly(every, startTime, endTime, timezone, from),
      daily: (every, time, timezone) =>
          _calculateDaily(every, time, timezone, from),
      weekly: (weekdays, time, timezone) =>
          _calculateWeekly(weekdays, time, timezone, from),
      monthlyFixed: (day, time, timezone) =>
          _calculateMonthlyFixed(day, time, timezone, from),
      monthlyNth: (weekOfMonth, weekday, time, timezone) =>
          _calculateMonthlyNth(weekOfMonth, weekday, time, timezone, from),
      yearlyFixed: (month, day, time, timezone) =>
          _calculateYearlyFixed(month, day, time, timezone, from),
      yearlyNth: (month, weekOfMonth, weekday, time, timezone) =>
          _calculateYearlyNth(month, weekOfMonth, weekday, time, timezone, from),
    );
  }

  static List<DateTime> getUpcomingOccurrences(
      RecurrenceRule rule, DateTime from, int count) {
    final result = <DateTime>[];
    var cursor = from;
    for (var i = 0; i < count; i++) {
      cursor = calculateNextDue(rule, cursor);
      result.add(cursor);
    }
    return result;
  }

  static bool isDueToday(DateTime nextDueAt) {
    final now = DateTime.now().toUtc();
    final local = nextDueAt.toLocal();
    final today = DateTime.now();
    return local.year == today.year &&
        local.month == today.month &&
        local.day == today.day;
  }

  static bool isOverdue(DateTime nextDueAt) {
    return nextDueAt.isBefore(DateTime.now().toUtc());
  }

  // ---------------------------------------------------------------------------
  // Private calculators
  // ---------------------------------------------------------------------------

  static DateTime _calculateHourly(int every, String startTime,
      String? endTime, String timezone, DateTime from) {
    final location = tz.getLocation(timezone);
    final tzFrom = tz.TZDateTime.from(from, location);
    final start = _parseTime(startTime);
    final end = endTime != null ? _parseTime(endTime) : (hour: 24, minute: 0);

    var candidate = tzFrom.add(const Duration(minutes: 1));

    // If past endTime, jump to next day's startTime
    if (candidate.hour > end.hour ||
        (candidate.hour == end.hour && candidate.minute >= end.minute)) {
      candidate = tz.TZDateTime(
          location,
          candidate.year,
          candidate.month,
          candidate.day + 1,
          start.hour,
          start.minute);
    }
    // If before startTime, jump to startTime today
    if (candidate.hour < start.hour ||
        (candidate.hour == start.hour && candidate.minute < start.minute)) {
      candidate = tz.TZDateTime(location, candidate.year, candidate.month,
          candidate.day, start.hour, start.minute);
    }

    // Align to next slot boundary
    final minutesSinceStart =
        (candidate.hour - start.hour) * 60 + candidate.minute - start.minute;
    final slotIndex = minutesSinceStart ~/ (every * 60);
    final nextSlotMinutes = (slotIndex + 1) * every * 60;
    final nextHour = start.hour + nextSlotMinutes ~/ 60;
    final nextMinute = start.minute + nextSlotMinutes % 60;

    if (nextHour > end.hour ||
        (nextHour == end.hour && nextMinute >= end.minute)) {
      return tz.TZDateTime(location, candidate.year, candidate.month,
          candidate.day + 1, start.hour, start.minute);
    }

    return tz.TZDateTime(location, candidate.year, candidate.month,
        candidate.day, nextHour, nextMinute);
  }

  static DateTime _calculateDaily(
      int every, String time, String timezone, DateTime from) {
    final location = tz.getLocation(timezone);
    final tzFrom = tz.TZDateTime.from(from, location);
    final t = _parseTime(time);

    final todaySlot = tz.TZDateTime(
        location, tzFrom.year, tzFrom.month, tzFrom.day, t.hour, t.minute);

    if (todaySlot.isAfter(tzFrom)) {
      return todaySlot;
    }
    return tz.TZDateTime(location, tzFrom.year, tzFrom.month,
        tzFrom.day + every, t.hour, t.minute);
  }

  static DateTime _calculateWeekly(List<String> weekdays, String time,
      String timezone, DateTime from) {
    final location = tz.getLocation(timezone);
    final tzFrom = tz.TZDateTime.from(from, location);
    final t = _parseTime(time);
    final targetDays = weekdays.map(_weekdayFromString).toSet();

    for (var i = 0; i <= 7; i++) {
      final candidate = tz.TZDateTime(location, tzFrom.year, tzFrom.month,
          tzFrom.day + i, t.hour, t.minute);
      if (candidate.isAfter(tzFrom) && targetDays.contains(candidate.weekday)) {
        return candidate;
      }
    }
    // Should not reach here for valid rules
    return tzFrom.add(const Duration(days: 7));
  }

  static DateTime _calculateMonthlyFixed(
      int day, String time, String timezone, DateTime from) {
    final location = tz.getLocation(timezone);
    final tzFrom = tz.TZDateTime.from(from, location);
    final t = _parseTime(time);

    // Try this month, clamped to last day
    for (var delta = 0; delta <= 2; delta++) {
      final month = tzFrom.month + delta;
      final year = tzFrom.year + (month - 1) ~/ 12;
      final adjustedMonth = ((month - 1) % 12) + 1;
      final lastDay = _lastDayOfMonth(year, adjustedMonth);
      final actualDay = day > lastDay ? lastDay : day;
      final candidate = tz.TZDateTime(
          location, year, adjustedMonth, actualDay, t.hour, t.minute);
      if (candidate.isAfter(tzFrom)) return candidate;
    }
    return tzFrom.add(const Duration(days: 32));
  }

  static DateTime _calculateMonthlyNth(int weekOfMonth, String weekday,
      String time, String timezone, DateTime from) {
    final location = tz.getLocation(timezone);
    final tzFrom = tz.TZDateTime.from(from, location);
    final t = _parseTime(time);
    final targetWeekday = _weekdayFromString(weekday);

    for (var delta = 0; delta <= 3; delta++) {
      final month = tzFrom.month + delta;
      final year = tzFrom.year + (month - 1) ~/ 12;
      final adjustedMonth = ((month - 1) % 12) + 1;
      final nthDate = _nthWeekday(year, adjustedMonth, weekOfMonth, targetWeekday);
      if (nthDate == null) continue;
      final candidate = tz.TZDateTime(
          location, nthDate.year, nthDate.month, nthDate.day, t.hour, t.minute);
      if (candidate.isAfter(tzFrom)) return candidate;
    }
    return tzFrom.add(const Duration(days: 32));
  }

  static DateTime _calculateYearlyFixed(
      int month, int day, String time, String timezone, DateTime from) {
    final location = tz.getLocation(timezone);
    final tzFrom = tz.TZDateTime.from(from, location);
    final t = _parseTime(time);

    for (var delta = 0; delta <= 1; delta++) {
      final year = tzFrom.year + delta;
      final lastDay = _lastDayOfMonth(year, month);
      final actualDay = day > lastDay ? lastDay : day;
      final candidate = tz.TZDateTime(
          location, year, month, actualDay, t.hour, t.minute);
      if (candidate.isAfter(tzFrom)) return candidate;
    }
    return tzFrom.add(const Duration(days: 366));
  }

  static DateTime _calculateYearlyNth(int month, int weekOfMonth,
      String weekday, String time, String timezone, DateTime from) {
    final location = tz.getLocation(timezone);
    final tzFrom = tz.TZDateTime.from(from, location);
    final t = _parseTime(time);
    final targetWeekday = _weekdayFromString(weekday);

    for (var delta = 0; delta <= 1; delta++) {
      final year = tzFrom.year + delta;
      final nthDate = _nthWeekday(year, month, weekOfMonth, targetWeekday);
      if (nthDate == null) continue;
      final candidate = tz.TZDateTime(
          location, nthDate.year, nthDate.month, nthDate.day, t.hour, t.minute);
      if (candidate.isAfter(tzFrom)) return candidate;
    }
    return tzFrom.add(const Duration(days: 366));
  }
}
```

```dart
class TasksRepositoryImpl implements TasksRepository {
  TasksRepositoryImpl({required FirebaseFirestore firestore})
      : _firestore = firestore;

  final FirebaseFirestore _firestore;
  static const _uuid = Uuid();

  CollectionReference<Map<String, dynamic>> _tasksCol(String homeId) =>
      _firestore.collection('homes').doc(homeId).collection('tasks');

  @override
  Stream<List<Task>> watchHomeTasks(String homeId) {
    return _tasksCol(homeId)
        .where('status', whereIn: ['active', 'frozen'])
        .snapshots()
        .map((snap) => snap.docs.map(TaskModel.fromFirestore).toList());
  }

  @override
  Future<Task> fetchTask(String homeId, String taskId) async {
    final doc = await _tasksCol(homeId).doc(taskId).get();
    if (!doc.exists) throw ServerException('Task $taskId not found');
    return TaskModel.fromFirestore(doc);
  }

  @override
  Future<String> createTask(
      String homeId, TaskInput input, String createdByUid) async {
    final taskId = _uuid.v4();
    final nextDueAt = RecurrenceCalculator.calculateNextDue(
      input.recurrenceRule,
      DateTime.now().toUtc(),
    );
    final recurrenceType = TaskModel.recurrenceTypeFromRule(input.recurrenceRule);
    final currentAssigneeUid =
        input.assignmentOrder.isEmpty ? null : input.assignmentOrder.first;

    final data = TaskModel.inputToFirestore(
      input,
      homeId: homeId,
      createdByUid: createdByUid,
      nextDueAt: nextDueAt,
      recurrenceType: recurrenceType,
      currentAssigneeUid: currentAssigneeUid,
    );

    await _tasksCol(homeId).doc(taskId).set(data);
    return taskId;
  }

  @override
  Future<void> updateTask(
      String homeId, String taskId, TaskInput input) async {
    final nextDueAt = RecurrenceCalculator.calculateNextDue(
      input.recurrenceRule,
      DateTime.now().toUtc(),
    );
    final recurrenceType = TaskModel.recurrenceTypeFromRule(input.recurrenceRule);

    await _tasksCol(homeId).doc(taskId).update({
      'title': input.title.trim(),
      'description': input.description,
      'visualKind': input.visualKind,
      'visualValue': input.visualValue,
      'recurrenceType': recurrenceType,
      'recurrenceRule': TaskModel.inputToFirestore(
        input,
        homeId: homeId,
        createdByUid: '',
        nextDueAt: nextDueAt,
        recurrenceType: recurrenceType,
      )['recurrenceRule'],
      'assignmentMode': input.assignmentMode,
      'assignmentOrder': input.assignmentOrder,
      'difficultyWeight': input.difficultyWeight,
      'nextDueAt': Timestamp.fromDate(nextDueAt),
      'updatedAt': FieldValue.serverTimestamp(),
    });
  }

  @override
  Future<void> freezeTask(String homeId, String taskId) async {
    await _tasksCol(homeId).doc(taskId).update({
      'status': 'frozen',
      'updatedAt': FieldValue.serverTimestamp(),
    });
  }

  @override
  Future<void> unfreezeTask(String homeId, String taskId) async {
    await _tasksCol(homeId).doc(taskId).update({
      'status': 'active',
      'updatedAt': FieldValue.serverTimestamp(),
    });
  }

  @override
  Future<void> deleteTask(
      String homeId, String taskId, String deletedByUid) async {
    final batch = _firestore.batch();

    // Mark task as deleted
    batch.update(_tasksCol(homeId).doc(taskId), {
      'status': 'deleted',
      'updatedAt': FieldValue.serverTimestamp(),
    });

    // Audit event
    final eventRef = _firestore
        .collection('homes')
        .doc(homeId)
        .collection('task_events')
        .doc();
    batch.set(eventRef, {
      'type': 'task_deleted',
      'taskId': taskId,
      'deletedByUid': deletedByUid,
      'timestamp': FieldValue.serverTimestamp(),
    });

    await batch.commit();
  }

  @override
  Future<void> reorderAssignees(
      String homeId, String taskId, List<String> order) async {
    await _tasksCol(homeId).doc(taskId).update({
      'assignmentOrder': order,
      'currentAssigneeUid': order.isEmpty ? null : order.first,
      'updatedAt': FieldValue.serverTimestamp(),
    });
  }
}
```

```dart
class _AllTasksScreenState extends ConsumerState<AllTasksScreen> {
  TaskStatus _filter = TaskStatus.active;
  String? _assigneeFilter;

  @override
  Widget build(BuildContext context) {
    final l10n = AppLocalizations.of(context);
    final homeAsync = ref.watch(currentHomeProvider);
    final authState = ref.watch(authProvider);
    final uid = authState.whenOrNull(authenticated: (u) => u.uid) ?? '';

    return homeAsync.when(
      loading: () => Scaffold(
        appBar: AppBar(title: Text(l10n.tasks_all_title)),
        body: const LoadingWidget(),
      ),
      error: (_, __) => Scaffold(
        appBar: AppBar(title: Text(l10n.tasks_all_title)),
        body: Center(child: Text(l10n.error_generic)),
      ),
      data: (home) {
        if (home == null) {
          return Scaffold(
            appBar: AppBar(title: Text(l10n.tasks_all_title)),
            body: Center(child: Text(l10n.error_generic)),
          );
        }

        final membershipsAsync = uid.isNotEmpty
            ? ref.watch(userMembershipsProvider(uid))
            : null;
        final memberships = membershipsAsync?.valueOrNull ?? [];
        final myMembership = memberships
            .where((m) => m.homeId == home.id)
            .cast<HomeMembership?>()
            .firstOrNull;
        final myRole = myMembership?.role;
        final canCreate =
            myRole == MemberRole.owner || myRole == MemberRole.admin;

        final tasksAsync = ref.watch(homeTasksProvider(home.id));

        return Scaffold(
          appBar: AppBar(
            title: Text(l10n.tasks_all_title),
            bottom: PreferredSize(
              preferredSize: const Size.fromHeight(48),
              child: _FilterBar(
                current: _filter,
                onChanged: (f) => setState(() => _filter = f),
              ),
            ),
          ),
          body: tasksAsync.when(
            loading: () => const LoadingWidget(),
            error: (_, __) => Center(child: Text(l10n.error_generic)),
            data: (allTasks) {
              var tasks = allTasks.where((t) => t.status == _filter).toList();
              if (_assigneeFilter != null) {
                tasks = tasks
                    .where((t) => t.currentAssigneeUid == _assigneeFilter)
                    .toList();
              }
              tasks.sort((a, b) => a.nextDueAt.compareTo(b.nextDueAt));

              if (tasks.isEmpty) {
                return Center(
                    key: const Key('tasks_empty_state'),
                    child: Text(l10n.tasks_empty));
              }

              return ListView.builder(
                key: const Key('tasks_list'),
                itemCount: tasks.length,
                itemBuilder: (_, i) {
                  final task = tasks[i];
                  return Dismissible(
                    key: Key('dismissible_${task.id}'),
                    background: _FreezeBackground(l10n: l10n),

                      if (direction == DismissDirection.startToEnd) {
                        await _toggleFreeze(context, task, home.id, l10n);
                        return false;
                      } else {
                        return _confirmDelete(context, l10n);
                      }
                    },
                    onDismissed: (direction) async {
                      if (direction == DismissDirection.endToStart) {
                        await ref
                            .read(tasksRepositoryProvider)
                            .deleteTask(home.id, task.id, uid);
                      }
                    },
                    child: TaskCard(
                      task: task,
                      onTap: () => context.go(
                          '/task/${task.id}'),
                    ),
                  );
                },
              );
            },
          ),
          floatingActionButton: canCreate
              ? FloatingActionButton(
                  key: const Key('create_task_fab'),
                  tooltip: l10n.tasks_create,

                  child: const Icon(Icons.add),
                )
              : null,
        );
      },
    );
  }

  Future<void> _toggleFreeze(BuildContext context, Task task, String homeId,
      AppLocalizations l10n) async {
    final repo = ref.read(tasksRepositoryProvider);
    if (task.status == TaskStatus.active) {
      await repo.freezeTask(homeId, task.id);
    } else {
      await repo.unfreezeTask(homeId, task.id);
    }
  }

  Future<bool> _confirmDelete(
      BuildContext context, AppLocalizations l10n) async {
    final result = await showDialog<bool>(
```

```dart
    return l10n.error_generic;

      BuildContext context, AppLocalizations l10n, String uid) async {
    final form = ref.read(taskFormProvider);
    final homeAsync = ref.read(currentHomeProvider);
    final home = homeAsync.valueOrNull;
    if (home == null) return;

    final input = ref.read(taskFormProvider.notifier).toInput();
    if (input == null) return;

    // Count current active tasks for validation
    final activeTasks = await ref
        .read(tasksRepositoryProvider)
        .watchHomeTasks(home.id)
        .first;
    final activeCount =
        activeTasks.where((t) => t.status.name == 'active').length;

    final result =
        TaskValidator.validate(input, home.limits, activeCount);
    if (!result.isOk) {
      ref.read(taskFormProvider.notifier).state =
          form.copyWith(errorCode: result.errorCode);
      return;
    }

    ref.read(taskFormProvider.notifier).state =
        form.copyWith(isSubmitting: true, errorCode: null);

    try {
      final repo = ref.read(tasksRepositoryProvider);
      if (isEditing) {
        await repo.updateTask(home.id, widget.editTaskId!, input);
      } else {
        await repo.createTask(home.id, input, uid);
      }
      if (context.mounted) Navigator.of(context).pop();
    } catch (_) {
      ref.read(taskFormProvider.notifier).state =
          form.copyWith(isSubmitting: false, errorCode: 'generic');
    }
  }
```

```dart
// lib/features/tasks/presentation/task_detail_screen.dart
import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:go_router/go_router.dart';
import 'package:intl/intl.dart';

import '../../../core/constants/routes.dart';
import '../../../l10n/app_localizations.dart';
import '../../../shared/widgets/loading_widget.dart';
import '../../auth/application/auth_provider.dart';
import '../../homes/application/current_home_provider.dart';
import '../../homes/application/homes_provider.dart';
import '../../homes/domain/home_membership.dart';
import '../application/recurrence_provider.dart';
import '../application/tasks_provider.dart';
import '../domain/task_status.dart';

class TaskDetailScreen extends ConsumerWidget {
  const TaskDetailScreen({super.key, required this.taskId});
  final String taskId;

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final l10n = AppLocalizations.of(context);
    final homeAsync = ref.watch(currentHomeProvider);
    final authState = ref.watch(authProvider);
    final uid = authState.whenOrNull(authenticated: (u) => u.uid) ?? '';

    return homeAsync.when(
      loading: () => Scaffold(
        appBar: AppBar(title: Text(l10n.tasks_detail_title)),
        body: const LoadingWidget(),
      ),
      error: (_, __) => Scaffold(
        appBar: AppBar(title: Text(l10n.tasks_detail_title)),
        body: Center(child: Text(l10n.error_generic)),
      ),
      data: (home) {
        if (home == null) {
          return Scaffold(
            appBar: AppBar(title: Text(l10n.tasks_detail_title)),
            body: Center(child: Text(l10n.error_generic)),
          );
        }

        final tasksAsync = ref.watch(homeTasksProvider(home.id));
        final membershipsAsync = uid.isNotEmpty
            ? ref.watch(userMembershipsProvider(uid))
            : null;
        final memberships = membershipsAsync?.valueOrNull ?? [];
        final myMembership = memberships
            .where((m) => m.homeId == home.id)
            .cast<HomeMembership?>()
            .firstOrNull;
        final myRole = myMembership?.role;
        final canEdit =
            myRole == MemberRole.owner || myRole == MemberRole.admin;

        return tasksAsync.when(
          loading: () => Scaffold(
            appBar: AppBar(title: Text(l10n.tasks_detail_title)),
            body: const LoadingWidget(),
          ),
          error: (_, __) => Scaffold(
            appBar: AppBar(title: Text(l10n.tasks_detail_title)),
            body: Center(child: Text(l10n.error_generic)),
          ),
          data: (tasks) {
            final task =
                tasks.where((t) => t.id == taskId).cast<dynamic>().firstOrNull;
            if (task == null) {
              return Scaffold(
                appBar: AppBar(title: Text(l10n.tasks_detail_title)),
                body: Center(child: Text(l10n.error_generic)),
              );
            }

            final upcoming = ref
                .watch(upcomingOccurrencesProvider(task.recurrenceRule));

            return Scaffold(
              appBar: AppBar(
                title: Text(task.title),
                actions: [
                  if (canEdit)
                    IconButton(
                      key: const Key('edit_task_button'),
                      icon: const Icon(Icons.edit),
                      onPressed: () =>
                          context.go('/tasks/$taskId/edit'),
                    ),
                ],
              ),
              body: ListView(
                padding: const EdgeInsets.all(16),
                children: [
                  // Visual + title
                  Row(
                    children: [
                      if (task.visualKind == 'emoji')
                        Text(task.visualValue,
                            style: const TextStyle(fontSize: 48))
                      else
                        const Icon(Icons.task_alt, size: 48),
                      const SizedBox(width: 16),
                      Expanded(
                        child: Column(
                          crossAxisAlignment: CrossAxisAlignment.start,
                          children: [
                            Text(task.title,
                                style: Theme.of(context).textTheme.headlineSmall),
                            if (task.description != null)
                              Text(task.description!),
                          ],
                        ),
                      ),
                    ],
                  ),
                  const Divider(height: 32),

                  // Status chip
                  if (task.status == TaskStatus.frozen)
                    const Chip(
                      key: Key('frozen_chip'),
                      label: Text('Congelada'),
                      avatar: Icon(Icons.pause_circle_outline),
                    ),

                  // Current assignee
                  ListTile(
                    key: const Key('current_assignee_tile'),
                    leading: const Icon(Icons.person),
                    title: Text(l10n.tasks_assignee_current),
                    subtitle: Text(
                        task.currentAssigneeUid ?? l10n.tasks_no_assignee),
                  ),

                  // Next due
                  ListTile(
                    key: const Key('next_due_tile'),
                    leading: const Icon(Icons.schedule),
                    title: Text(l10n.tasks_next_occurrence),
                    subtitle: Text(
                      DateFormat.yMMMd().add_Hm().format(
                          task.nextDueAt.toLocal()),
                    ),
                  ),

                  // Difficulty
                  ListTile(
                    key: const Key('difficulty_tile'),
                    leading: const Icon(Icons.fitness_center),
                    title: Text(l10n.tasks_difficulty_value(
                        task.difficultyWeight.toStringAsFixed(0))),
                  ),

                  const Divider(height: 32),

                  // Upcoming occurrences
                  Text(l10n.tasks_next_occurrence,
                      style: Theme.of(context).textTheme.titleSmall),
                  const SizedBox(height: 8),
                  ...upcoming.take(3).map((d) => ListTile(
                        dense: true,
                        title: Text(
                          DateFormat.yMMMd().add_Hm().format(d.toLocal()),
                        ),
                      )),
                ],
              ),
            );
          },
        );
      },
    );
  }
}
```
